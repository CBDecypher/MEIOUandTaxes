country_event = {
	id = DG_Catholics.001
	title = "DG_Catholics.001.title"
	desc = "DG_Catholics.001.desc"
	picture = all_church_state

	major = yes
	fire_only_once = yes
	
	trigger = {
		is_religion_enabled = protestant
		religion_years = { protestant = 50 }
		tag = PAP
	}
	
	mean_time_to_happen = {
		years = 10
	}
	
	immediate = {
		hidden_effect = {
			set_global_flag = DG_Counter_Reformation
			random_list = {
				10 = { 
					set_global_flag = DG_Small_Counterref 
					log = "DG:[GetYear]:2530:Counter_Reformation:Birth:Rome:[This.GetName]:Small Counterreformation"
				}
				80 = { 
					set_global_flag = DG_Mild_Counterref 
					log = "DG:[GetYear]:2530:Counter_Reformation:Birth:Rome:[This.GetName]:Mild Counterreformation"
				}
				10 = { 
					set_global_flag = DG_Large_Counterref 
					log = "DG:[GetYear]:2530:Counter_Reformation:Birth:Rome:[This.GetName]:Large Counterreformation"
				}
			}
		}
		set_country_flag = DG_Catholic_Revival
	}
	
	option = {
		name = "DG_Catholics.001.opta"
		hidden_effect = {
			random_known_country = {
				limit = {
					religion = catholic
					NOT = { has_country_flag = DG_Catholic_Revival }
					NOT = { has_country_flag = DG_Religious_Compromise }
				}
				country_event = { id = DG_Catholics.002 days = 45 random = 15 }
			}
		}
	}
}

	is_triggered_only = yes
	
	option = {
		name = "DG_Catholics.091.opta"
		log = "DG:[GetYear]:[Root.GetName]:refused legate request to give Rome back to [PAP.GetName]:DG_Catholics.091"
		log = "DG:[GetYear]:[Root.GetName]:was excommunicated:DG_Catholics.091"
		if = {
			limit = { religion = catholic }
			excommunicate = ROOT
			add_papal_influence = -200
		}
		2530 = { add_permanent_claim = PAP }
		custom_tooltip = urbs_mundi_opinion_held_tt
		hidden_effect = {
			every_known_country = {
				limit = { religion = catholic OR = { government = theocracy patriarch_authority = 0.75 } }
				reverse_add_opinion = { who = ROOT modifier = urbs_mundi_unlawful_strong }
			}
			every_known_country = {
				limit = { religion = catholic NOT = { government = theocracy } NOT = { patriarch_authority = 0.75 } }
				reverse_add_opinion = { who = ROOT modifier = urbs_mundi_unlawful_weak }
			}
		}
		PAP = { country_event = { id = DG_Catholics.092 } }
	}
	option = {
		name = "DG_Catholics.091.optb"
		log = "DG:[GetYear]:[Root.GetName]:accepted legate request and gave Rome back to [PAP.GetName]:DG_Catholics.091"
		if = {
			limit = { NOT = { exists = PAP } }
			PAP = {
				clr_country_flag = POP_Init
			}
		}
		2530 = {
			add_core = PAP
			cede_province = PAP
		}
		custom_tooltip = urbs_mundi_opinion_good_tt
		hidden_effect = {
			every_known_country = {
				limit = { religion = catholic }
				reverse_add_opinion = { who = ROOT modifier = urbs_mundi_good }
			}
		}
		hidden_effect = {
			PAP = {
				country_event = { id = Init.1 }
				if = {
					limit = {
						NOT = {
							religion = catholic
						}
					}
					change_religion = catholic
				}
				add_government_reform = Pol_Mediumship4
				add_government_reform = Pol_Organisation3
				add_government_reform = Pol_Nobility1
				add_government_reform = Pol_Assembly1
			}
		}
	}
}
