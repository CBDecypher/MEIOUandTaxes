hoftag_agenda_trigger = {
	emperor = {
		OR = {
			OR = {
				has_country_flag = Hft_Agen_Trib_1
				has_country_flag = Hft_Agen_Trib_2
				has_country_flag = Hft_Agen_Trib_3
			}
			
	
			OR = {
				has_country_flag = Hft_Agen_Consc_1
				has_country_flag = Hft_Agen_Consc_2
				has_country_flag = Hft_Agen_Consc_3
			}
			
	
			OR = {
				has_country_flag = Hft_Agen_Centr_1
				has_country_flag = Hft_Agen_Centr_2
				has_country_flag = Hft_Agen_Centr_3
			}
			
			has_country_flag = Hft_Agen_Reichstag_1
			has_country_flag = Hft_Agen_Defense_1
			has_country_flag = Hft_Agen_Romzug_1
		}
	}
}

### Meta Script
Hoftag_Add_Agenda_Topic_Trigger = {
	NOT = {
		has_country_modifier = Imp_$topic$_CD
		OR = {
			has_country_flag = Hft_Agen_$topic$_1
			has_country_flag = Hft_Agen_$topic$_2
			has_country_flag = Hft_Agen_$topic$_3
		}
	}
	NOT = {	ERG = { check_key = { lhs = Hft_Agen value = 4 } } }
	OR = {
		adm = 3
		dip = 5
		mil = 4
	}		
}

HRE_Centralization_Trigger = {
	[[amount]
		ERG = { check_key = { lhs = HRE_Centralization value = $amount$ }	}
	]
	[[variable]
		ERG = { 
			set_key = { lhs = $variable$ which = PREV } 
			check_key = { lhs = HRE_Centralization which = $variable$ }
			set_key = { lhs = $variable$ value = 0 }
		}
	]
}
