mil_tech_at_war_start = {
	if = { 		limit = { mil_tech = 50 } 	set_country_flag = mil_tech_50 }
	else_if = { limit = { mil_tech = 40 }  	set_country_flag = mil_tech_40 }
	else_if = { limit = { mil_tech = 30 }  	set_country_flag = mil_tech_30 }
	else_if = { limit = { mil_tech = 20 } 	set_country_flag = mil_tech_20 }
	else_if = { limit = { mil_tech = 10 }  	set_country_flag = mil_tech_10 }
	else_if = { limit = { mil_tech = 0 }	set_country_flag = mil_tech_0 }	
}
