	is_triggered_only = yes # See cb_types.1
	months = 12000
	ai_peace_desire = -30

	prerequisites = {
		OR = {
			AND = {
				has_idea = exploration_ideas_6
	
				NOT = {
					has_country_flag = NewWorldTag
				}
			}
			AND = {
				is_colonial_nation = yes
				overlord = {
					NOT = {
						has_country_flag = NewWorldTag
					}
				}
			}
		}
		
		FROM = {
			has_country_flag = NewWorldTag
			
			is_colonial_nation = no
			
			NOT = {
				has_institution = Banking
			}
		}
		
		is_revolutionary = no
	}

	war_goal = new_world_conquest
}
